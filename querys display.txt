6) عرض التحليلات

أكثر الأفلام تأجيرًا

SELECT
    f.title,
    SUM(r.rental_count) AS total_rentals
FROM fact_rental r
JOIN dim_film f
ON r.film_key = f.film_key
GROUP BY f.title
ORDER BY total_rentals DESC
LIMIT 10;
